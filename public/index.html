my-agent.ts
import { Varuna } from 'varuna';

const guardian = new Varuna({
  wallet: agentWallet,
  strategy: {
    autonomy: 'full-auto',
    yieldAwareProtection: true
  }
});

await guardian.start();  // That's it!

// Listen for events
guardian.on('alert', ({ riskLevel }) => {
  console.log(`Risk: ${riskLevel}`);
});

guardian.on('protection', ({ result }) => {
  console.log(`Protected! HF: ${result.newHF}`);
});
